// Inject the service
private clipboardService = inject(ClipboardService);

// Copy text
async copyText() {{ '{' }}
  const success = await this.clipboardService.copyToClipboard('text');
  console.log('Copy success:', success);
{{ '}' }}

// Read clipboard
async readClipboard() {{ '{' }}
  const content = await this.clipboardService.readFromClipboard();
  console.log('Clipboard content:', content);
{{ '}' }}
Usage Instructions
Basic Usage:
<button appCopyToClipboard="Text to copy">Copy</button>
With Custom Messages:
<button 
  appCopyToClipboard="Text to copy"
  copySuccessMessage="Custom success!"
  copyErrorMessage="Custom error!">
  Copy
</button>
Copy Element Content:
<div appCopyToClipboard>
  This text will be copied
</div>
Disable Tooltip:
<button 
  appCopyToClipboard="Text"
  [showTooltip]="false">
  Silent Copy
</button>
Features:
  • Modern Clipboard API with fallback
  • Custom success/error messages
  • Animated tooltips
  • Copy animation effects
  • Works with any element
